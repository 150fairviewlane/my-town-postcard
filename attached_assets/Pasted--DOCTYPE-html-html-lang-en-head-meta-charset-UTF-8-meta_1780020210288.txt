// ADMIN — SVG MAP

function renderMap(){
  const state = document.getElementById('adminStateSelect').value;
  const territories = TERRITORY_DB[state] || [];
  const svg = document.getElementById('adminMap');

  // Build county → territory lookup
  const countyMap = {};
  territories.forEach(t=>{
    t.counties.forEach(c=>{ countyMap[c] = t; });
  });

  svg.innerHTML = '';

  // Render each county as a rounded rect (simplified — in production use real GeoJSON paths)
  GA_COUNTIES.forEach(county=>{
    const territory = countyMap[county.name];
    let fillClass = '';
    if(territory){
      fillClass = `territory-${territory.status}`;
      if(territory.id === selectedTerritoryId) fillClass += ' highlight';
    }

    const g = document.createElementNS('http://www.w3.org/2000/svg','g');

    // County shape (rounded rect as placeholder — replace with real GeoJSON in production)
    const rect = document.createElementNS('http://www.w3.org/2000/svg','rect');
    rect.setAttribute('x', county.cx - county.w/2);
    rect.setAttribute('y', county.cy - county.h/2);
    rect.setAttribute('width', county.w);
    rect.setAttribute('height', county.h);
    rect.setAttribute('rx','3');
    rect.setAttribute('class',`county-path ${fillClass}`);
    rect.setAttribute('data-county', county.name);
    rect.setAttribute('data-territory-id', territory?.id || '');

    // County label
    const text = document.createElementNS('http://www.w3.org/2000/svg','text');
    text.setAttribute('x', county.cx);
    text.setAttribute('y', county.cy + 3);
    text.setAttribute('text-anchor','middle');
    text.setAttribute('font-size','7');
    text.setAttribute('font-family','DM Sans, sans-serif');
    text.setAttribute('fill','rgba(0,0,0,0.5)');
    text.setAttribute('pointer-events','none');
    text.textContent = county.name;

    g.appendChild(rect);
    g.appendChild(text);

    // Event listeners
    rect.addEventListener('mousemove', e=>showTooltip(e, county, territory));
    rect.addEventListener('mouseleave', hideTooltip);
    rect.addEventListener('click', ()=>{ if(territory) selectTerritory(territory.id); });

    svg.appendChild(g);
  });

  // State border label
  const label = document.createElementNS('http://www.w3.org/2000/svg','text');
  label.setAttribute('x','20');
  label.setAttribute('y','20');
  label.setAttribute('font-size','11');

  label.setAttribute('fill','rgba(0,0,0,0.3)');
  label.setAttribute('letter-spacing','2');
  label.textContent = 'GEORGIA';
  svg.appendChild(label);
}

function showTooltip(e, county, territory){
  const tt = document.getElementById('mapTooltip');
  const container = tt.parentElement.getBoundingClientRect();
  const x = e.clientX - container.left + 12;
  const y = e.clientY - container.top - 10;

  document.getElementById('ttName').textContent = county.name + ' County';
  document.getElementById('ttTerritory').textContent = territory ? territory.name : 'Unassigned';

  const statusEl = document.getElementById('ttStatus');
  if(territory){
    statusEl.textContent = territory.status.charAt(0).toUpperCase() + territory.status.slice(1);
    statusEl.className = `tt-status tt-${territory.status}`;
  } else {
    statusEl.textContent = 'Not in any territory';
    statusEl.className = 'tt-status tt-unassigned';
  }

  tt.style.left = Math.min(x, container.width - 230) + 'px';
  tt.style.top = Math.max(0, y) + 'px';
  tt.classList.add('visible');
}

function searchCounties(){
  const q = document.getElementById('countySearch').value.toLowerCase();
  const sugg = document.getElementById('countySuggestions');
  const state = document.getElementById('adminStateSelect').value;

  // Get all counties in territory DB for this state to check used ones
  const usedCounties = new Set();
  (TERRITORY_DB[state]||[]).forEach(t=>t.counties.forEach(c=>usedCounties.add(c)));

  const matches = GA_COUNTIES
    .filter(c=> c.name.toLowerCase().includes(q) && q.length > 0)
    .slice(0,8);

  if(!matches.length){ sugg.classList.remove('visible'); return; }

  sugg.innerHTML = matches.map(c=>{
    const inNew   = newTerritoryCounties.includes(c.name);
    const inExist = usedCounties.has(c.name);
    const disabled = inNew || inExist;
    return `<div class="county-sugg-item${disabled?' in-territory':''}"
      onclick="${disabled?'':`addChip('${c.name}')`}"
      title="${inNew?'Already added':inExist?'In another territory':'Add '+c.name}">
      ${c.name} County${inNew?' ✓':inExist?' (taken)':''}
    </div>`;
  }).join('');
  sugg.classList.add('visible');
}

function saveNewTerritory(){

  const households = parseInt(document.getElementById('ntfHouseholds').value) || 0;

  if(!name){ showToast('Please enter a territory name'); return; }
  if(newTerritoryCounties.length < 1){ showToast('Please add at least one county'); return; }
  if(households < 5000){ showToast('Minimum 5,000 households required for 4 mailing zones'); return; }

  const state = document.getElementById('adminStateSelect').value;
  if(!TERRITORY_DB[state]) TERRITORY_DB[state] = [];

  const newId = state + '-' + String(TERRITORY_DB[state].length + 1).padStart(3,'0');
  TERRITORY_DB[state].push({
    id: newId,
    name,
    state,
    counties: [...newTerritoryCounties],
    households,
    status: 'available',
    dealer: null,
  });

  toggleNewForm();
  renderSidebar();
  showToast(`Territory "${name}" saved!`);

  // In Replit: POST /api/territories { name, state, counties, households }
}

            onchange="onTerritoryPick('${t.id}')">

            <div class="topt-indicator" style="background:${t.status==='available'?'#86efac':t.status==='taken'?'#fca5a5':'#fde68a'}"></div>
            <div class="topt-info">
              <div class="topt-name">${t.name}</div>
              <div class="topt-counties">${t.counties.join(' · ')}</div>
              <div class="topt-status status-${t.status}">${t.status==='available'?'✓ Available':t.status==='taken'?'✗ Taken':'⏳ Pending'}</div>
            </div>
          </label>
        </div>`;
      }).join('')
    : `<div style="font-size:12px;color:var(--ink-light);text-align:center;padding:20px">No territories available in this state yet.</div>`;

  selectedDealerTerritoryId = null;
  document.getElementById('submitBtn').disabled = true;
}

function onTerritoryPick(id){
  selectedDealerTerritoryId = id;
  document.getElementById('submitBtn').disabled = false;
}

// DEALER SIGNUP — SUBMIT

function submitSignup(){
  const name    = document.getElementById('dealerName').value.trim();
  const email   = document.getElementById('dealerEmail').value.trim();
  const phone   = document.getElementById('dealerPhone').value.trim();

  if(!name||!email||!phone){ showToast('Please fill in your name, email, and phone'); return; }
  if(!selectedDealerTerritoryId){ showToast('Please select a territory'); return; }

  const state = document.getElementById('dealerStateSelect').value;
  const t = (TERRITORY_DB[state]||[]).find(t=>t.id===selectedDealerTerritoryId);
  if(!t||t.status==='taken'){ showToast('This territory is no longer available'); return; }

  // Mark as pending locally
  t.status = 'pending';
  t.dealer = name;

  // In Replit: POST /api/dealer-signup { name, email, phone, territoryId }
  // Then redirect to Stripe checkout for $99/month subscription

  document.getElementById('signupForm').style.display = 'none';
  const success = document.getElementById('successPanel');
  success.classList.add('visible');
  document.getElementById('successMsg').innerHTML =
    `<strong>${name}</strong>, you've claimed the <strong>${t.name}</strong> territory!<br><br>
     We'll be in touch at <strong>${email}</strong> to complete your setup and activate your account.<br><br>
     Your territory covers: ${t.counties.join(', ')} ${t.counties.length===1?'County':'Counties'} with approximately ${t.households.toLocaleString()} households across 4 mailing zones.`;

  renderSidebar(); // Update admin map to show pending
  showToast('Territory claimed — welcome aboard!');
}

function showToast(msg){
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(()=>t.classList.remove('show'), 2800);
}
