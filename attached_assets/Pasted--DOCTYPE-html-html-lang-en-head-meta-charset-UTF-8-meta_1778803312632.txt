// ── STATE ──────────────────────────────────────────────────────
let logoB64 = null;
let photoB64 = null;
let resultImageUrl = null;

// ── MENU ──────────────────────────────────────────────────────
function addMenuItem(val=''){
  const list = document.getElementById('menuList');
  if(list.children.length >= 4) return;
  const row = document.createElement('div');
  row.className = 'mrow';
  row.innerHTML = `<input type="text" placeholder="Item Name $Price" value="${val}" oninput="buildPrompt()">
    <button class="rm-btn" onclick="this.parentElement.remove();buildPrompt()" title="Remove">×</button>`;
  list.appendChild(row);
  buildPrompt();
}

function getMenu(){
  return Array.from(document.querySelectorAll('.mrow input'))
    .map(i=>i.value.trim()).filter(Boolean).slice(0,4);
}

// ── UPLOAD ────────────────────────────────────────────────────
function handleUpload(input, previewId, zoneId, varName){
  const file = input.files[0]; if(!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    if(varName==='logoB64') logoB64 = e.target.result;

// ── COPY PROMPT ───────────────────────────────────────────────
async function copyPrompt(){
  if(!window._rawPrompt) return;
  try {
    await navigator.clipboard.writeText(window._rawPrompt);
    const btn = document.getElementById('copyBtn');
    btn.textContent = '✓ Copied!';
    btn.classList.add('copied');
    showToast('Prompt copied to clipboard!');
    setTimeout(()=>{ btn.textContent='📋 Copy Prompt'; btn.classList.remove('copied'); }, 2500);
  } catch(e){
    // Fallback for browsers that block clipboard
    const ta = document.createElement('textarea');
    ta.value = window._rawPrompt;
    ta.style.position='fixed';ta.style.opacity='0';
    document.body.appendChild(ta);
    ta.select();document.execCommand('copy');
    document.body.removeChild(ta);
    showToast('Prompt copied!');
  }
}

// ── LAUNCH AI ─────────────────────────────────────────────────
async function launchAI(){
  const d = getData();
  if(!d.bizName){ alert('Please enter your business name first.'); return; }

  // Copy prompt to clipboard first
  await copyPrompt();

  // Determine URL
  // Claude supports a ?q= param that pre-fills the input
  // ChatGPT supports a URL that opens a new chat (no pre-fill param available)
  const aiUrl = d.ai === 'claude'
    ? `https://claude.ai/new?q=${encodeURIComponent(window._rawPrompt.substring(0,2000))}`
    : 'https://chatgpt.com/';

  // Open popup sized to fit alongside the app
  const pw = Math.min(820, screen.width * 0.55);
  const ph = Math.min(900, screen.height * 0.9);
  const pl = window.screen.width - pw - 20;
  const pt = Math.max(0, (screen.height - ph) / 2);

  const popup = window.open(
    aiUrl,
    'ai_console',
    `width=${pw},height=${ph},left=${pl},top=${pt},resizable=yes,scrollbars=yes`
  );

  if(!popup){
    // Popup blocked
    alert('Popup blocked! Please allow popups for this site, then click Launch again.\n\nYour prompt has been copied to the clipboard.');
    return;
  }

  // Show result panel and update steps
  showResultPanel();
  setStep(3);

  // Show tip
  showToast('AI console opened — paste your prompt and hit send!');
}

function handleResultDrop(input){
  const file = input.files[0]; if(!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    resultImageUrl = e.target.result;
    const img = document.getElementById('dropResult');
    const zone = document.getElementById('dropZone');
    img.src = resultImageUrl;
    img.classList.add('visible');
    zone.classList.add('has-img');
    zone.querySelector('.drop-icon').style.display='none';
    zone.querySelector('.drop-title').style.display='none';
    zone.querySelector('.drop-sub').style.display='none';
    document.getElementById('infoStatus').textContent = '✓ Ad image received';
    document.getElementById('infoStatus').style.color = 'var(--green)';
    document.getElementById('infoActions').style.display='flex';
    setStep(4);
  };
  reader.readAsDataURL(file);
}

// Support paste anywhere on the page
document.addEventListener('paste', e => {
  const items = e.clipboardData?.items;
  if(!items) return;
  for(const item of items){
    if(item.type.startsWith('image/')){
      const file = item.getAsFile();
      const fakeInput = { files: [file] };
      handleResultDrop(fakeInput);
      break;
    }
  }
});
