function onIndustryChange(){
  const ind = document.getElementById('industry').value;
  if(!ind || !INDUSTRIES[ind]) return;
  const data = INDUSTRIES[ind];
  // Set tagline if empty
  const tg = document.getElementById('tagline');
  if(!tg.value) tg.value = data.taglines[0];
  // Populate menu items
  buildMenuFromArray(data.menu);
}

// ── MENU BUILDER ──────────────────────────────────────────────
function buildMenuFromArray(items){
  const list = document.getElementById('menuList');
  list.innerHTML = '';
  items.forEach(item => addMenuItem(item));
}

function addMenuItem(val=''){
  const list = document.getElementById('menuList');
  const row = document.createElement('div');
  row.className = 'menu-item-row';
  row.innerHTML = `
    <input type="text" placeholder="e.g. Chicken Biscuit $5.99" value="${val}">
    <button class="rm-btn" onclick="this.parentElement.remove()" title="Remove">×</button>
  `;
  list.appendChild(row);
}

function getMenuItems(){
  return Array.from(document.querySelectorAll('.menu-item-row input'))
    .map(i=>i.value.trim()).filter(Boolean);
}

// ── FILE UPLOAD ───────────────────────────────────────────────
function handleUpload(input, previewId, zoneId, varName){
  const file = input.files[0];
  if(!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    const b64 = e.target.result;
    if(varName==='logoB64') logoB64 = b64;
    if(varName==='photoB64') photoB64 = b64;
    const preview = document.getElementById(previewId);
    preview.src = b64;
    preview.style.display = 'block';
    // Shrink upload zone text
    const zone = document.getElementById(zoneId);
    zone.querySelector('.upload-icon').style.display = 'none';
    zone.querySelector('.upload-txt').textContent = '✓ Uploaded';
    zone.querySelector('.upload-sub').textContent = file.name.substring(0,20);
  };
  reader.readAsDataURL(file);
}

  // Update overlay
  document.getElementById('ov-address').textContent = [d.address, d.city].filter(Boolean).join(', ');
  document.getElementById('ov-phone').textContent = d.phone;
  document.getElementById('ov-website').textContent = d.website;

  setUI('loading');

  const steps = ['Crafting your design brief...','Sending to DALL·E 3...','Generating your ad...','Applying finishing touches...'];
  let si = 0;
  const el = document.getElementById('loadingStep');
  const iv = setInterval(()=>{ si=(si+1)%steps.length; if(el) el.textContent=steps[si]; }, 4000);

  try {
    const body = {
      model: 'dall-e-3',
      prompt: buildDallePrompt(d),
      n: 1,
      size: '1024x1024',
      quality: 'standard',
      response_format: 'url',
    };

    const res = await fetch('https://api.openai.com/v1/images/generations', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${apiKey}`,
      },
      body: JSON.stringify(body),
    });

    clearInterval(iv);

    if(!res.ok){
      const err = await res.json().catch(()=>({}));
      throw new Error(err.error?.message || `OpenAI API error ${res.status}`);
    }

    const data = await res.json();
    const imgUrl = data.data?.[0]?.url;
    if(!imgUrl) throw new Error('No image returned from DALL·E. Please try again.');

    lastImageUrl = imgUrl;

    const img = document.getElementById('dalle-img');
    img.onload = () => setUI('success');
    img.onerror = () => { throw new Error('Failed to load generated image.'); };
    img.src = imgUrl;

  } catch(err){
    clearInterval(iv);
    document.getElementById('errorMsg').textContent = err.message;
    setUI('error');
  }
}

// ── UI STATE ──────────────────────────────────────────────────
function setUI(state){
  const empty = document.getElementById('emptyState');
  const load  = document.getElementById('loadingState');
  const err   = document.getElementById('errorState');
  const img   = document.getElementById('dalle-img');
  const ovl   = document.getElementById('adOverlay');
  const strip = document.getElementById('costStrip');
  const regen = document.getElementById('regenBtn');
  const save  = document.getElementById('saveBtn');
  const use   = document.getElementById('useBtn');
  const gen   = document.getElementById('generateBtn');

  empty.classList.remove('hidden');
  load.classList.remove('active');
  err.classList.remove('active');
  img.classList.remove('visible');
  ovl.classList.remove('visible');
  strip.style.display = 'none';
  regen.disabled = save.disabled = use.disabled = true;
  gen.disabled = false;

  if(state==='loading'){
    empty.classList.add('hidden');
    load.classList.add('active');
    gen.disabled = true;
  } else if(state==='success'){
    empty.classList.add('hidden');
    img.classList.add('visible');
    ovl.classList.add('visible');
    strip.style.display = 'flex';
    regen.disabled = save.disabled = use.disabled = false;
  } else if(state==='error'){
    empty.classList.add('hidden');
    err.classList.add('active');
  }
}
