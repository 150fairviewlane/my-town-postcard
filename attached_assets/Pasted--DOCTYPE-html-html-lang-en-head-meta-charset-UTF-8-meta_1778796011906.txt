// ── INDUSTRY CHANGE ────────────────────────────────────────────
function onIndustryChange(){
  const ind = document.getElementById('industry').value;
  if(!ind) return;
  const data = INDUSTRIES[ind] || INDUSTRIES['Other Service'];

  // Always update tagline, menu, and coupon when industry changes
  document.getElementById('tagline').value = data.tagline;

  // Rebuild menu items
  buildMenu(data.menu);

  // Populate coupon defaults
  document.getElementById('offerAmount').value = data.offer.amount;
  document.getElementById('offerItem').value   = data.offer.item;
  document.getElementById('offerFine').value   = data.offer.fine;

  // Load photos from library (falls back to hardcoded)
  loadPhotos(ind, data.photos);

  renderAd();
}

// ── LOAD PHOTOS ────────────────────────────────────────────────
async function loadPhotos(industry, fallback){
  const grid = document.getElementById('imgGrid');
  grid.innerHTML = '<div class="img-msg">Loading photos...</div>';
  let photos = fallback || [];

  // In Replit: fetch from image library
  try {
    const res = await fetch('/api/image-library?industry=' + encodeURIComponent(industry));
    const data = await res.json();
    if(data.images && data.images.length)
      photos = data.images.map(i => i.image_url);
  } catch(e){ /* use fallback */ }

  currentPhotos = photos;
  if(!photos.length){
    grid.innerHTML = '<div class="img-msg">No photos yet for this industry — upload your own below, or add photos in the Admin Image Library.</div>';
    return;
  }

  grid.innerHTML = '';
  photos.slice(0,8).forEach((url,i)=>{
    const div = document.createElement('div');
    div.className = 'img-thumb' + (i===0?' selected':'');
    div.onclick = ()=>{ selectPhoto(url); document.querySelectorAll('.img-thumb').forEach(t=>t.classList.remove('selected')); div.classList.add('selected'); };
    div.innerHTML = `<img src="${url}" loading="lazy" alt="photo ${i+1}"><div class="sel-check">✓</div>`;
    grid.appendChild(div);
  });

  selectedPhotoUrl = photos[0];
  currentPhotoIndex = 0;
  document.getElementById('shuffleBtn').disabled = false;
  setHeroPhoto(selectedPhotoUrl);
  renderAd();
}

function selectPhoto(url){
  selectedPhotoUrl = url;
  setHeroPhoto(url);
  renderAd();
}

function shufflePhoto(){
  if(!currentPhotos.length) return;
  currentPhotoIndex = (currentPhotoIndex+1) % currentPhotos.length;
  const url = currentPhotos[currentPhotoIndex];
  selectedPhotoUrl = url;
  document.querySelectorAll('.img-thumb').forEach((t,i)=>t.classList.toggle('selected', i===currentPhotoIndex));
  setHeroPhoto(url);
  renderAd();
}

function setHeroPhoto(url){
  const wrap = document.getElementById('photoWrap');
  const img  = document.getElementById('photoImg');
  if(url){
    img.src = url;
    wrap.style.display = 'block';
  } else {
    wrap.style.display = 'none';
  }
}

// ── UPLOAD ─────────────────────────────────────────────────────
function handleUpload(input){
  const file = input.files[0]; if(!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    const url = e.target.result;
    document.getElementById('uploadPreview').src = url;
    document.getElementById('uploadPreview').style.display = 'block';
    document.querySelectorAll('.img-thumb').forEach(t=>t.classList.remove('selected'));
    selectedPhotoUrl = url;
    setHeroPhoto(url);
    renderAd();
  };
  reader.readAsDataURL(file);
}

function addMenuItem(val=''){
  const list = document.getElementById('menuList');
  if(list.children.length >= 4) return;
  const row = document.createElement('div');
  row.className = 'mrow';
  row.innerHTML = `<input type="text" placeholder="Item Name $Price" value="${val}" oninput="renderAd()">

  list.appendChild(row);
  renderAd();
}
function getMenu(){
  return Array.from(document.querySelectorAll('.mrow input')).map(i=>i.value.trim()).filter(Boolean).slice(0,4);
}
function parseMenuItem(str){
  const m = str.match(/^(.+?)\s*(\$[\d.]+.*)$/);
  return m ? {name:m[1].trim(), price:m[2].trim()} : {name:str, price:''};
}

// ── RENDER AD ──────────────────────────────────────────────────
function renderAd(){
  const ac = document.querySelector('input[name="color"]:checked')?.value || '#C8541A';

  // Show canvas
  document.getElementById('adWrap').classList.add('visible');
  document.getElementById('adEmpty').classList.add('hidden');
  document.getElementById('adWrap').style.setProperty('--ac', ac);

  // Business name
  const l1 = document.getElementById('bizLine1').value.trim();
  const l2 = document.getElementById('bizLine2').value.trim();
  document.getElementById('ovL1').textContent = l1;
  document.getElementById('ovL2').textContent = l2;
  document.getElementById('ovL2').style.color = ac;

  // Tagline
  const tg = document.getElementById('tagline').value.trim();
  document.getElementById('ovTgTxt').textContent = tg;
  document.getElementById('ovTgRule').style.background = `linear-gradient(90deg,${ac},transparent)`;

  // Menu
  const menuEl = document.getElementById('ovMenu');
  menuEl.innerHTML = getMenu().map(item=>{
    const p = parseMenuItem(item);
    return `<div class="menu-row">
      <span class="mi-name">${p.name}</span>
      ${p.price ? `<span class="mi-dots"></span><span class="mi-price" style="color:${ac}">${p.price}</span>` : ''}
    </div>`;
  }).join('');

  // Coupon
  const amt  = document.getElementById('offerAmount').value.trim();
  const item = document.getElementById('offerItem').value.trim();
  const fine = document.getElementById('offerFine').value.trim();
  const couponEl = document.getElementById('ovCoupon');
  if(amt || item){
    couponEl.style.display = 'flex';
    document.getElementById('ovCpAmt').textContent = amt;
    document.getElementById('ovCpItem').textContent = item;
    document.getElementById('ovCpItem').style.color = ac;
    document.getElementById('ovCpFine').textContent = fine;
  } else {
    couponEl.style.display = 'none';
  }

  // Phone + address
  const phone = document.getElementById('phone').value.trim();
  const addr  = document.getElementById('address').value.trim();
  const city  = document.getElementById('city').value.trim();
  document.getElementById('ovPhNum').textContent = phone;
  document.getElementById('ovPhAddr').textContent = [addr, city].filter(Boolean).join(', ');

  // QR label
  const qrUrl = document.getElementById('qrUrl').value.trim();
  document.getElementById('ovQrTxt').textContent = qrUrl || (phone ? 'Tap to call' : '');

  document.getElementById('useBtn').disabled = false;
}
