<div style={{ width: "100%" }}>
  <PostcardScaleContext.Provider value={postcardScale}>
    <div
      ref={gridRef}
      style={{
        width: "100%",
        aspectRatio: "12 / 9",
        display: "grid",
        gridTemplateColumns: "repeat(12, 1fr)",
        gridTemplateRows: "repeat(9, 1fr)",
        gap: 2,

        // 🔥 Makes it feel like a real postcard
        background: "#e5e7eb",
        padding: 6,
        borderRadius: 8,
        boxShadow:
          "0 20px 60px rgba(0,0,0,0.25), inset 0 0 0 2px #fff",
        transform: "rotateX(2deg)",
      }}
    >
      {sortedSpots.map((spot) => {
        const isSelected = selected?.id === spot.id;
        const isPaid =
          (spot.status === "paid" || spot.status === "reserved") &&
          spot.gridArea !== "mb";

        const sampleKey = SPOT_SAMPLE_MAP[spot.gridArea];
        const sampleContent =
          !isPaid && sampleKey
            ? getSampleAd(sampleKey, SIZE_MAP[spot.size] || "S")
            : null;

        const pos = GRID_POSITIONS[spot.gridArea];
        if (!pos) return null;

        return (
          <ScaledCell key={spot.id} pos={pos}>
            <div
              onClick={() => !isPaid && openCreator(spot)}
              style={{
                width: "100%",
                height: "100%",
                position: "relative",
                borderRadius: 6,
                overflow: "hidden",

                // ✨ Interaction polish
                cursor: isPaid ? "default" : "pointer",
                transition: "all 0.2s ease",
                boxShadow: isSelected
                  ? "0 0 0 3px #f59e0b inset"
                  : "0 4px 14px rgba(0,0,0,0.2)",

                transform: isSelected ? "scale(1.02)" : "scale(1)",
              }}
              onMouseEnter={(e) => {
                if (!isPaid) {
                  e.currentTarget.style.transform = "scale(1.04)";
                  e.currentTarget.style.boxShadow =
                    "0 10px 30px rgba(0,0,0,0.35)";
                }
              }}
              onMouseLeave={(e) => {
                if (!isPaid) {
                  e.currentTarget.style.transform = "scale(1)";
                  e.currentTarget.style.boxShadow =
                    "0 4px 14px rgba(0,0,0,0.2)";
                }
              }}
            >
              {isPaid ? (
                <PaidAd spot={spot} />
              ) : sampleContent ? (
                sampleContent
              ) : (
                <AvailableSpot
                  spot={spot}
                  isSelected={isSelected}
                  onClick={() => openCreator(spot)}
                />
              )}

              {/* 🔥 subtle overlay for realism */}
              <div
                style={{
                  position: "absolute",
                  inset: 0,
                  background:
                    "linear-gradient(to bottom, transparent 70%, rgba(0,0,0,0.15))",
                  pointerEvents: "none",
                }}
              />
            </div>
          </ScaledCell>
        );
      })}

      {/* Fixed cells (unchanged) */}
      {fixedAreas.map((area) => {
        const pos = GRID_POSITIONS[area];
        if (!pos) return null;

        return (
          <ScaledCell key={area} pos={pos} pointerEvents="none">
            {renderFixedCell(area)}
          </ScaledCell>
        );
      })}
    </div>
  </PostcardScaleContext.Provider>
</div>
