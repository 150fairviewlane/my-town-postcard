let map, geojsonLayer, userMarker, selectedTerritoryId = null;

function initMap(){
  map = L.map('map', { zoomControl:true }).setView([32.8, -83.4], 7);

  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution:'© <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a>',
    maxZoom:18,
  }).addTo(map);

  // Load Georgia county GeoJSON from US Census Bureau
  loadCountyGeoJSON();
}

async function loadCountyGeoJSON(){
  try {
    // US Census Bureau TIGER GeoJSON for Georgia counties (state FIPS 13)
    const url = 'https://raw.githubusercontent.com/plotly/datasets/master/geojson-counties-fips.json';
    const res = await fetch(url);
    const allCounties = await res.json();

    // Filter to Georgia only (FIPS starts with 13)
    const gaFeatures = allCounties.features.filter(f => f.id && f.id.startsWith('13'));

    const gaGeoJSON = { type:'FeatureCollection', features: gaFeatures };
    renderCountyLayer(gaGeoJSON);
  } catch(e) {
    console.warn('GeoJSON load failed, using marker-based display', e);
    // Fall back to circle markers if GeoJSON fails
    renderMarkersOnly();
  }
}

function renderCountyLayer(geojson){
  // Build county name to territory lookup
  const countyToTerritory = {};
  TERRITORIES.forEach(t => {
    t.counties.forEach(c => {
      if(!countyToTerritory[c.toLowerCase()]) countyToTerritory[c.toLowerCase()] = [];
      countyToTerritory[c.toLowerCase()].push(t);
    });
  });

  geojsonLayer = L.geoJSON(geojson, {
    style: feature => {
      const countyName = feature.properties.NAME;
      const territories = countyToTerritory[countyName.toLowerCase()] || [];
      // Use first territory's status for color (split counties show dominant status)
      const status = territories.length ? territories[0].status : 'unassigned';
      return getCountyStyle(status, false);
    },
    onEachFeature: (feature, layer) => {
      const countyName = feature.properties.NAME;
      const territories = (countyToTerritory[countyName.toLowerCase()] || []);

      layer.on({
        mouseover: e => {
          const l = e.target;
          l.setStyle({ weight:2.5, fillOpacity:0.85 });
          l.bringToFront();
          showMapTooltip(countyName, territories);
        },
        mouseout: e => {
          geojsonLayer.resetStyle(e.target);
          hideMapTooltip();
        },
        click: () => {
          if(territories.length === 1){
            selectTerritoryOnMap(territories[0]);
          } else if(territories.length > 1){
            // Multiple territories in this county (metro split) — show list
            showNearbyTerritories(territories, `${countyName} County — Multiple Territories`);
          }
        }
      });

      if(territories.length){
        layer.bindTooltip(
          `<div style="font-weight:700">${countyName} County</div>
           <div style="font-size:11px;color:#666">${territories.map(t=>t.name).join('<br>')}</div>`,
          {sticky:true, opacity:0.92}
        );
      }
    }
  }).addTo(map);
}

function getCountyStyle(status, highlighted){
  const styles = {
    available: { fillColor:'#86efac', fillOpacity:.55, color:'#16a34a', weight: highlighted?2:0.8 },
    taken:     { fillColor:'#fca5a5', fillOpacity:.55, color:'#dc2626', weight: highlighted?2:0.8 },
    pending:   { fillColor:'#fde68a', fillOpacity:.55, color:'#d97706', weight: highlighted?2:0.8 },
    selected:  { fillColor:'#bfdbfe', fillOpacity:.75, color:'#2563eb', weight:2.5 },
    unassigned:{ fillColor:'#d4c9ba', fillOpacity:.4,  color:'#9ca3af', weight:0.5 },
  };
  return styles[status] || styles.unassigned;
}

async function searchByZip(){
  const zip = document.getElementById('zipInput').value.trim();
  if(zip.length < 5){ showToast('Please enter a 5-digit zip code'); return; }

  // Look up county from zip
  const county = ZIP_TO_COUNTY[zip];

function findNearbyTerritories(lat, lng, homeCounty){
  // Calculate distance from user to each territory centroid
  const withDistance = TERRITORIES.map(t => ({
    ...t,
    distance: haversineDistance(lat, lng, t.lat, t.lng)
  }));

  // Sort by distance
  withDistance.sort((a,b) => a.distance - b.distance);

  // Take nearest 10, deduplicate by id
  const seen = new Set();
  const nearby = [];
  for(const t of withDistance){
    if(!seen.has(t.id)){
      seen.add(t.id);
      nearby.push(t);
    }
    if(nearby.length >= 10) break;
  }

  // Zoom map to show user location and nearby territories
  const bounds = L.latLngBounds([[lat, lng]]);
  nearby.forEach(t => bounds.extend([t.lat, t.lng]));
  map.fitBounds(bounds.pad(0.15));

  // Hide map tip
  document.getElementById('mapTip').classList.add('hidden');

  showNearbyTerritories(nearby, `Territories near ${homeCounty} County`);
}

function haversineDistance(lat1, lng1, lat2, lng2){
  const R = 3959; // miles
  const dLat = (lat2-lat1) * Math.PI/180;
  const dLng = (lng2-lng1) * Math.PI/180;
  const a = Math.sin(dLat/2)**2 + Math.cos(lat1*Math.PI/180)*Math.cos(lat2*Math.PI/180)*Math.sin(dLng/2)**2;
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
}
